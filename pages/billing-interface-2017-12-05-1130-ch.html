7生成发票

#request
{
  "accid": "n_1023",    //产品（广告商）id
  "alltotal": 200.00,   //总金额
  "body": "Click Tech Limited", //主体
  "contid": "2",     //合同ID
  "currency": "M",   //币种
  "data": [{
    "DESCRIPTION": "Jan 1st to Jan 31st",
    "ITEM": "dfd",
    "NOTE": "",
    "PAYOUT": "",
    "QTY": "",
    "TOTAL": 200
  }],
  "empid": "1234",    //上传发票人IDU8->怎么获取-->
  "fpdate": "201711",  //业务周期, //目前格式是1512576000
  "inid": "1234"       //发票NO
}

8发票状态查询

#request
{
  "inid":"1234"
}

#return
{
  "code":10001,
  "msg":"成功",
  "data":{
    "currency": "", //货币
    "paidAmount": "0",
    "status": "0",   //0,1-审核, 2-复审  0 可以删除 其他不可以删除 
    "totalAmount": "0"
  }
}
